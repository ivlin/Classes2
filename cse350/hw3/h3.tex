\begin{enumerate}

    \begin{tabular}{|l|l|l|l|}
        \hline
        Original String & Prefix Border & & Suffix Border\\
        \hline
        abracadabra& abra & cad & abra \\
        \hline
        abracadabr& abr & acad &abr\\
        \hline
        abracadab& & abracadab& \\
        \hline
        abracada& a&bracad&a \\
        \hline
        abracad& & abracad & \\
        \hline
        abraca& a & brac & a \\
        \hline
        abrac& & abrac & \\
        \hline
        abra& a & br & a \\
        \hline
        abr& & abr & \\
        \hline
        ab& & ab & \\
        \hline
        a& & a & \\
        \hline
    \end{tabular}

  \item[(b)]
    Construct the Knuth-Morris-Pratt mismatch table for $P$.

      \hline
    \end{tabular}
